81  void AxisBinning::Set(const std::string &binning)
82  {
84  if(!binning.size()) return;
85  int ncommas{0};
86  for(auto& c : binning) if(c==',') ++ncommas;
87  if(ncommas!=2) {
88  Error::PrintWarning(Form("Binning string \"%s\" has to contain 2 commas", binning.c_str()));
89  return;
90  }
92  int nbins = 0;
93  double width = 0.;
94  double from = 0.;
95  double to = 0.;
97  TString line(binning.c_str());
98  TString tok;
99  UChar_t sw = 0;
100  Ssiz_t start{0};
101  while(line.Tokenize(tok, start, ",")) {
102  switch(sw) {
103  case 0 :
104  if(tok.Contains('.') || tok.Contains('e') || tok.Contains('E')) width = tok.Atof();
105  else nbins = tok.Atoi();
106  break;
107  case 1 : from = tok.Atof(); break;
108  case 2 : to = tok.Atof(); break;
109  default :
110  Error::PrintFatalError("AxisBinning::Set(const std::string&)");
111  std::terminate();
112  }
113  ++sw;
114  }
116  if(nbins) Set(nbins, from, to); // if first argument is int
117  else Set(width, from, to); // if first argument is double
118  }
