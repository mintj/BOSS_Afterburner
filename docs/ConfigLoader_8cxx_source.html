131  size_t ConfigLoader::LoadConfiguration(const std::string &filename)
132  {
134  std::ifstream file(filename);
135  if(!file.is_open()) {
136  std::cout << "WARNING: Could not load configuration file \"" << filename << "\"" << std::endl;
137  return -1;
138  }
140  std::cout << std::endl << "LOADING CONFIGURATION FROM \"" << filename << "\"" << std::endl;
142  std::string line;
143  while(getline(file, line)) {
146  if(line.back()<' ') line.pop_back();
148  if(line.find('=') == std::string::npos && line.find(' ') == std::string::npos) continue;
150  String::Trim(line);
152  if(String::IsComment(line)) continue;
153  ConfigParBase* par = nullptr;
157  if(line.find('=') == std::string::npos) par = ConfigParBase::GetParameter(line);
159  else if(line.back() == '=' || line.back() == '{') {
160  line.pop_back();
161  String::Trim(line);
162  par = GetParameter(line);
163  }
165  if(line.find('{') == std::string::npos) {
166  par = GetParameter(line);
167  line = line.substr(line.find_first_of('=')+1);
168  String::Trim(line);
169  par->AddValue(line);
170  continue;
171  }
173  else {
174  par = GetParameter(line);
175  line = line.substr(line.find_first_of('{')+1);
176  String::Trim(line);
177  ImportValues(par, line);
178  }
181  while(getline(file, line)) {
184  String::Trim(line);
186  if(line.find('}') == std::string::npos) ImportValues(par, line);
188  else {

192  }
