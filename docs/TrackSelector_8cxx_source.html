386  void TrackSelector::CreateChargedCollection()
387  {
390  if(!fCreateChargedCollection) return;
391 
393  fGoodChargedTracks.clear();
394  fPIDInstance = ParticleID::instance();
395 
397  if(!fEvtRecEvent->totalCharged())
398 
400  fNetChargeMDC = 0;
401  fLog << MSG::DEBUG << "Starting 'good' charged track selection:" << endmsg;
402  for(int i = 0; i < fEvtRecEvent->totalCharged(); ++i) {
405 
406  // * Get track info from Main Drift Chamber
407  fLog << MSG::DEBUG << " charged track " << i << "/" << fEvtRecEvent->totalCharged() << endmsg;
408  fTrackIterator = fEvtRecTrkCol->begin() + i;
409  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
410  ++fCounter_Nmdcvalid;
411  fTrackMDC = (*fTrackIterator)->mdcTrack();
412 
413  // * Get kinematics of track
414  double phi = fTrackMDC->helix(1);
415  double vr =
416  (fTrackMDC->x() - fVertexPoint.x()) * cos(phi) +
417  (fTrackMDC->y() - fVertexPoint.y()) * sin(phi);
418 
419  // * Get radii of track vertex
420  HepVector a = fTrackMDC->helix();
421  HepSymMatrix Ea = fTrackMDC->err();
422  HepPoint3D point0(0., 0., 0.); // initial point for MDC reconstruction
423  VFHelix helixip(point0, a, Ea);
424  helixip.pivot(fVertexPoint);
425  HepVector vecipa = helixip.a();
426  double rvxy = fabs(vecipa[0]); // nearest distance to IP in xy plane
427  double rvz = vecipa[3]; // nearest distance to IP in z direction
428  double rvphi = vecipa[1]; // angle in the xy-plane (?)
429 
432  if(fCut_Vz .FailsMax(fabs(fTrackMDC->z()))) continue;
433  if(fCut_Vxy.FailsMax(vr) ) continue;
434  if(fCut_Rz .FailsMax(fabs(rvz)) ) continue;
435  if(fCut_Rxy.FailsMax(rvxy) ) continue;
436  if(fCut_CosTheta.FailsMax(fabs(cos(fTrackMDC->theta())))) continue;
437 
439  // * Add charged track to vector
440  fGoodChargedTracks.push_back(*fTrackIterator);
441  fNetChargeMDC += fTrackMDC->charge();
442 
444  if(fNTuple_charged.DoWrite()) {
445  fNTuple_charged.GetItem<double>("vx") = fTrackMDC->x();
446  fNTuple_charged.GetItem<double>("vy") = fTrackMDC->y();
447  fNTuple_charged.GetItem<double>("vz") = fTrackMDC->z();
448  fNTuple_charged.GetItem<double>("vr") = vr;
449  fNTuple_charged.GetItem<double>("rvxy") = rvxy;
450  fNTuple_charged.GetItem<double>("rvz") = rvz;
451  fNTuple_charged.GetItem<double>("rvphi") = rvphi;
452  fNTuple_charged.GetItem<double>("phi") = phi;
453  fNTuple_charged.GetItem<double>("p") = fTrackMDC->p();
454  fNTuple_charged.Write();
455  }
456 
458  WriteDedxInfo(*fTrackIterator, fNTuple_dedx);
459 
461  if(fNTuple_TofEC.DoWrite() || fNTuple_TofIB.DoWrite() || fNTuple_TofOB.DoWrite()) {
462  // * Check if MDC and TOF info for track are valid *
463  if(!(*fTrackIterator)->isMdcTrackValid()) continue;
464  if(!(*fTrackIterator)->isTofTrackValid()) continue;
465  // * Get momentum as determined by MDC *
466  fTrackMDC = (*fTrackIterator)->mdcTrack();
467  double ptrk;
468  if(fTrackMDC) ptrk = fTrackMDC->p();
469  SmartRefVector<RecTofTrack> tofTrkCol = (*fTrackIterator)->tofTrack();
470  SmartRefVector<RecTofTrack>::iterator iter_tof = tofTrkCol.begin();
471  for(; iter_tof != tofTrkCol.end(); ++iter_tof) {
472  TofHitStatus hitStatus;
473  hitStatus.setStatus((*iter_tof)->status());
474  if(!hitStatus.is_counter()) continue;
475  if(hitStatus.is_barrel()) {
476  if(hitStatus.layer() == 1) { // inner barrel
477  WriteTofInformation(iter_tof, ptrk, fNTuple_TofIB);
478  } else if(hitStatus.layer() == 2) { // outer barrel
479  WriteTofInformation(iter_tof, ptrk, fNTuple_TofOB);
480  }
481  }
482  else if(hitStatus.layer() == 0) // end cap
483  WriteTofInformation(iter_tof, ptrk, fNTuple_TofEC);
484  }
485  }
486 
488  }
489 
490  fLog << MSG::DEBUG << "Number of 'good' charged tracks: " << fGoodChargedTracks.size() << endmsg;
492  }

768  void TrackSelector::WriteTofInformation(SmartRefVector<RecTofTrack>::iterator iter_tof, double ptrk, NTupleContainer &tuple)
769  {

775  std::vector<double> texp(gNMasses);
776  for(size_t j = 0; j < texp.size(); ++j) {
777  double gb = ptrk/gMasses[j]; // v = p/m (non-relativistic velocity)
778  double beta = gb/sqrt(1+gb*gb);
779  texp[j] = 10 * path /beta/gSpeedOfLight; // hypothesis ToF

781

784  tuple.GetItem<double>("tof") = (*iter_tof)->tof();

789  tuple.GetItem<double>("qual") = (*iter_tof)->quality();
790  tuple.GetItem<double>("tof_e") = path - texp[0];
