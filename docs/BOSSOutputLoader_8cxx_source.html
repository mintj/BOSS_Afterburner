28  BOSSOutputLoader::BOSSOutputLoader(const char* path, bool print_branches, bool print_averages) :
29  fDirectoryPath(path)
30  {
33  if(fDirectoryPath.EndsWith(".root")) {
36  TFile file(fDirectoryPath.Data());

39  TString fname(fDirectoryPath);
40  auto pos = fDirectoryPath.Last('/');
41  fDirectoryPath.Resize(pos);
42  fname.Remove(0, pos+1);
44  fFileNames.push_back(fname);
46  }
48  else {
51  if(fDirectoryPath.EndsWith("/")) fDirectoryPath.Chop();

56  if(!files) {
57  std::cout << "ERROR: Directory \"" << fDirectoryPath << "\" does not exist" << std::endl;
58  return;

61  TIter fileIter(files);
62  TObject *obj = nullptr;
63  while ((obj = fileIter())) {
64  auto systemFile = dynamic_cast<TSystemFile*>(obj);

72  }
